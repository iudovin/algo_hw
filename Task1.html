def g(n):
    s = 0
    b = 0
    while b * b < n:                    # n   
        for i in range(b):              # b 
            for j in range(0, i, 2):    # i/2 
                s += 1
            j = 1
            while j < n:                # log(n) 
                s += 1
                j *= 2
        b += 1
    return s
